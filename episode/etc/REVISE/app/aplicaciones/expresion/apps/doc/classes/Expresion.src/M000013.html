  def evaluar1op()
    return false if @fin

    @operadores.each do |op|
      @expr_vector.each_index do |i|
         if (@expr_vector[i].valor==op)
           salida = @operador.calcular(i,@expr_vector,@traza)
           return true if salida
         end
      end
    end
    
    return false
  end
